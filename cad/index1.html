import socket import time import os import json import asyncio import threading import sqlite3 from datetime import datetime import websockets # ================================================== # ⚙️ CONFIGURATION # ================================================== H7_IP = "192.168.0.100" H7_PORT = 502 UNIT_ID = 1 LOOP_DELAY = 0.02 SOCKET_TIMEOUT = 0.5 BASE_DIR = os.path.dirname(os.path.abspath(__file__)) DB_FILE = os.path.join(BASE_DIR, "smd_packing_analytics.db") # ── 📋 ประกาศลำดับรายชื่อ STATES ตามผังใหม่ของนิว 100% ── STATES = [ 'LOAD_CARRIER', 'INDEX_CARRIER', 'POWER_ON', 'SET_PARAMS', 'SENSOR_CHECK_CARRIER', 'READY', 'LOAD_PART', 'VISION', 'CHECK_TEMP', 'FEED_CARRIER', 'CAMERA1_COUNT', 'ENCODER_COUNT', 'COUNT_CHECK', 'TARGET_CHECK', 'SEAL', 'VISION_QC', 'TAKEUP_REEL', 'ALARM' ] system_data = { "current_state": "READY", "running": False, "mode": "auto", "step_allowed": False, "cycles": 0, "speed_mul": 1.0, "ip0": 0, "ip1": 0, "op0": 0, "op1": 0, "station_id": "MODBUS-EXT-IO", "fw_version": "STANDARD-v2.0", "camera1_count": 0, "encoder_count": 0, "target_count": 5 # กำหนด Target จำลองไว้ที่ 5 ชิ้นต่อ 1 ม้วนใหญ่ } connected_clients = set() trigger_reset_timer = False # ================================================== # 🗄️ SQLITE DATABASE FUNCTIONS # ================================================== def init_db(): conn = sqlite3.connect(DB_FILE) cursor = conn.cursor() cursor.execute(""" CREATE TABLE IF NOT EXISTS machine_logs ( id INTEGER PRIMARY KEY AUTOINCREMENT, timestamp TEXT NOT NULL, state_name TEXT NOT NULL, duration_sec REAL NOT NULL, control_mode TEXT NOT NULL, current_cycle INTEGER NOT NULL, status TEXT NOT NULL ) """) cursor.execute(""" CREATE TABLE IF NOT EXISTS production_summary ( id INTEGER PRIMARY KEY AUTOINCREMENT, end_timestamp TEXT NOT NULL, cycle_number INTEGER NOT NULL, total_duration_sec REAL NOT NULL ) """) conn.commit() conn.close() print(f"🗄️ [SQL Database] Initialized successfully. File: {DB_FILE}") def log_state_transition_to_sql(state_name, duration_sec, mode, cycles, status="NORMAL"): try: conn = sqlite3.connect(DB_FILE) cursor = conn.cursor() now_str = datetime.now().strftime('%Y-%m-%d %H:%M:%S.%f')[:-3] cursor.execute(""" INSERT INTO machine_logs (timestamp, state_name, duration_sec, control_mode, current_cycle, status) VALUES (?, ?, ?, ?, ?, ?) """, (now_str, state_name, round(duration_sec, 3), mode, cycles, status)) conn.commit() conn.close() print(f"💾 [SQL Saved] State: {state_name} | Spent: {round(duration_sec, 2)}s | Status: {status}") except Exception as e: print(f"❌ DB Error (Transition): {e}") def log_cycle_complete_to_sql(cycle_number, total_duration_sec): try: conn = sqlite3.connect(DB_FILE) cursor = conn.cursor() now_str = datetime.now().strftime('%Y-%m-%d %H:%M:%S') cursor.execute(""" INSERT INTO production_summary (end_timestamp, cycle_number, total_duration_sec) VALUES (?, ?, ?) """, (now_str, cycle_number, round(total_duration_sec, 2))) conn.commit() conn.close() print(f"🏆 [SQL Production Summary] Cycle {cycle_number} saved! Total loop time: {round(total_duration_sec, 2)}s") except Exception as e: print(f"❌ DB Error (Production): {e}") def get_historical_logs(): try: conn = sqlite3.connect(DB_FILE) cursor = conn.cursor() cursor.execute(""" SELECT state_name, duration_sec, current_cycle, control_mode, timestamp FROM machine_logs ORDER BY id DESC LIMIT 100 """) rows = cursor.fetchall() conn.close() history = [] for r in reversed(rows): history.append({ "state": r[0], "duration": r[1], "cycle": r[2], "mode": r[3], "time": r[4] }) return history except Exception as e: print(f"❌ DB Read Error: {e}") return [] # ================================================== # 🔌 STANDARD MODBUS CLIENT # ================================================== class SCHClient: def __init__(self, ip, port): self.ip = ip self.port = port self.sock = None def connect(self): try: self.sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM) self.sock.settimeout(SOCKET_TIMEOUT) self.sock.connect((self.ip, self.port)) print("✅ [Modbus] Connected to H7 STM32 Successfully") return True except: self.sock = None return False def send(self, data: bytes): try: if self.sock: self.sock.sendall(data) except: self.sock = None def write_op(self, bank, value): if not self.sock: return try: addr = 64 + 8 * bank pkt = b'\x00\x00\x00\x00\x00\x08' + UNIT_ID.to_bytes(1,'big') + b'\x0F' + addr.to_bytes(2,'big') + b'\x00\x08' + b'\x01' + value.to_bytes(1,'big') self.send(pkt) self.sock.recv(12) except: self.sock = None # ================================================== # 🧠 NEW FSM MAIN BRAIN LOOP (ตามผังใหม่ของนิว) # ================================================== def fsm_brain_loop(): global system_data, trigger_reset_timer io = SCHClient(H7_IP, H7_PORT) state_index = STATES.index('READY') last_transition_time = time.time() cycle_start_time = time.time() # กำหนดฐานเวลาจำลองของแต่ละสเตปตามผังใหม่ durations = { 'LOAD_CARRIER': 0.5, 'INDEX_CARRIER': 1.0, 'POWER_ON': 0.5, 'SET_PARAMS': 0.5, 'SENSOR_CHECK_CARRIER': 0.4, 'READY': 0.5, 'LOAD_PART': 0.8, 'VISION': 0.4, 'CHECK_TEMP': 0.4, 'FEED_CARRIER': 1.0, 'CAMERA1_COUNT': 0.3, 'ENCODER_COUNT': 0.3, 'COUNT_CHECK': 0.5, 'TARGET_CHECK': 0.4, 'SEAL': 1.2, 'VISION_QC': 0.4, 'TAKEUP_REEL': 1.0, 'ALARM': 0.5 } # จุดตรวจสอบสำหรับโหมด Semi-Auto (เฉพาะประเภท Decision ในผัง) checkpoints = ['SENSOR_CHECK_CARRIER', 'VISION', 'CHECK_TEMP', 'COUNT_CHECK', 'TARGET_CHECK', 'VISION_QC'] while True: if io.sock is None: io.connect() try: now = time.time() if trigger_reset_timer: state_index = STATES.index(system_data["current_state"]) last_transition_time = now cycle_start_time = now trigger_reset_timer = False curr = STATES[state_index] system_data["current_state"] = curr if system_data["running"] and curr != 'ALARM': target_dur = durations.get(curr, 0.5) / system_data["speed_mul"] # ลอจิกหยุดรอการตัดสินใจในโหมด Semi-Auto if curr in checkpoints and system_data["mode"] == "semi" and not system_data["step_allowed"] and (now - last_transition_time >= target_dur): system_data["step_allowed"] = True elif not system_data["step_allowed"] and (now - last_transition_time >= target_dur): actual_spent = now - last_transition_time log_state_transition_to_sql(curr, actual_spent, system_data["mode"], system_data["cycles"]) # ── 🔀 ลอจิกการเปลี่ยนทิศทางสเตปตามทางแยกในผังใหม่ ── if curr == 'CAMERA1_COUNT': system_data["camera1_count"] += 1 state_index = STATES.index('ENCODER_COUNT') # บังคับวิ่งไปนับเอนโค้ดเดอร์ต่อ elif curr == 'ENCODER_COUNT': system_data["encoder_count"] += 1 state_index = STATES.index('COUNT_CHECK') elif curr == 'COUNT_CHECK': # ตรวจสอบค่า Mismatch ระหว่างกล้องและเอนโค้ดเดอร์ if system_data["camera1_count"] == system_data["encoder_count"]: state_index = STATES.index('TARGET_CHECK') else: state_index = STATES.index('ALARM') log_state_transition_to_sql(curr, 0, system_data["mode"], system_data["cycles"], "🚨 MISMATCH ALARM DETECTED") elif curr == 'TARGET_CHECK': # เช็คว่ายอดปั๊มครบจำนวนชิ้นงานตามเป้าหรือยัง (count == target?) if system_data["camera1_count"] >= system_data["target_count"]: state_index = STATES.index('SEAL') else: # ถ้ายังไม่ครบ ให้ลากลูปกลับไปที่ FEED_CARRIER เพื่อปั๊มชิ้นถัดไป state_index = STATES.index('FEED_CARRIER') elif curr == 'TAKEUP_REEL': # ครบลูปการม้วนเก็บ 1 ม้วนใหญ่ บันทึกสถิติรวมของ Cycle นั้น total_cycle_time = now - cycle_start_time system_data["cycles"] += 1 log_cycle_complete_to_sql(system_data["cycles"], total_cycle_time) # รีเซ็ตตัวนับย่อยเพื่อเริ่มม้วนใหม่ system_data["camera1_count"] = 0 system_data["encoder_count"] = 0 cycle_start_time = now # วนลูปกลับไปเริ่มที่ SENSOR_CHECK_CARRIER ตามผังใหม่ state_index = STATES.index('SENSOR_CHECK_CARRIER') else: # ขยับไปตามลำดับสเตปปกติ state_index = (state_index + 1) % len(STATES) last_transition_time = now # ── 🔄 IO MAPPING (ถอดรหัสรวบบิต I/O ส่งสัญญาณลงบอร์ดฮาร์ดแวร์จริง) ── out_motor_index = 0 out_motor_roll2 = 0 out_seal_cyl = 0 out_feed_cyl = 0 out_alarm_light = 0 ip0_mask = 0 if system_data["running"]: if curr == 'INDEX_CARRIER': out_motor_index = 1 elif curr == 'FEED_CARRIER': out_feed_cyl = 1 elif curr == 'SEAL': out_seal_cyl = 1 elif curr == 'TAKEUP_REEL': out_motor_roll2 = 1 elif curr == 'ALARM': out_alarm_light = 1 # จำลองสถานะเซนเซอร์อินพุตแสดงบนบอร์ด if curr == 'SENSOR_CHECK_CARRIER': ip0_mask |= (1 << 0) elif curr == 'LOAD_PART': ip0_mask |= (1 << 1) elif curr == 'VISION': ip0_mask |= (1 << 2) elif curr == 'CHECK_TEMP': ip0_mask |= (1 << 3) elif curr == 'VISION_QC': ip0_mask |= (1 << 4) if curr == 'ALARM': out_alarm_light = 1 OP0 = 0x00 if out_motor_index == 1: OP0 |= 0x01 if out_motor_roll2 == 1: OP0 |= 0x02 if out_seal_cyl == 1: OP0 |= 0x04 if out_alarm_light == 1: OP0 |= 0x08 if out_feed_cyl == 1: OP0 |= 0x10 system_data["op0"] = OP0 system_data["ip0"] = ip0_mask if io.sock: io.write_op(0, OP0) io.write_op(1, 0x00) time.sleep(LOOP_DELAY) except Exception as e: io.sock = None time.sleep(0.5) # ================================================== # 🌐 WEBSOCKET BRIDGE SERVER # ================================================== async def ws_handler(websocket): global system_data, trigger_reset_timer print("🌐 Web Twin Connected!") connected_clients.add(websocket) try: async for message in websocket: data = json.loads(message) action = data.get("action") if action == "START": system_data["running"] = True system_data["current_state"] = "LOAD_CARRIER" system_data["cycles"] = 0 system_data["camera1_count"] = 0 system_data["encoder_count"] = 0 system_data["step_allowed"] = False trigger_reset_timer = True print("▶ START COMMAND RECEIVED") elif action == "RESET": system_data["running"] = False system_data["current_state"] = "READY" system_data["cycles"] = 0 system_data["camera1_count"] = 0 system_data["encoder_count"] = 0 system_data["step_allowed"] = False trigger_reset_timer = True print("↺ RESET COMMAND RECEIVED") elif action == "MODE": system_data["mode"] = data.get("mode", "auto") print(f"🕹️ MODE SWITCHED TO: {system_data['mode']}") elif action == "SPEED": system_data["speed_mul"] = float(data.get("value", 1.0)) elif action == "GET_HISTORY": history_data = get_historical_logs() await websocket.send(json.dumps({ "type": "HISTORY_RESPONSE", "data": history_data })) elif action == "DECISION": is_passed = data.get("value", True) curr_st = system_data["current_state"] if system_data["step_allowed"]: if is_passed: system_data["step_allowed"] = False trigger_reset_timer = True # ลอจิกจำลองข้ามสเตปแบบคุมแมนนวลกรณีเข้าเงื่อนไข Decision ทางแยก state_idx = STATES.index(system_data["current_state"]) if curr_st == 'COUNT_CHECK': system_data["current_state"] = 'TARGET_CHECK' elif curr_st == 'TARGET_CHECK': system_data["current_state"] = 'SEAL' if system_data["camera1_count"] >= system_data["target_count"] else 'FEED_CARRIER' else: system_data["current_state"] = STATES[(state_idx + 1) % len(STATES)] print(f"✅ STATE APPROVED: {system_data['current_state']}") else: system_data["current_state"] = "ALARM" log_state_transition_to_sql(curr_st, 0.0, system_data["mode"], system_data["cycles"], status="🚨 CRITICAL HUMAN NG STOP") system_data["step_allowed"] = False print(f"🚨 NG DETECTED! -> MACHINE ALARM") except websockets.exceptions.ConnectionClosed: print("🌐 Web Twin Disconnected") finally: connected_clients.remove(websocket) async def broadcast_state(): while True: if connected_clients: try: msg = json.dumps({ "type": "LIVE_SYNC", "system": system_data }) await asyncio.gather(*[client.send(msg) for client in connected_clients], return_exceptions=True) except: pass await asyncio.sleep(0.05) async def main_async(): init_db() threading.Thread(target=fsm_brain_loop, daemon=True).start() asyncio.create_task(broadcast_state()) async with websockets.serve(ws_handler, "0.0.0.0", 8765): print("🚀 [Web Server] WebSocket Running on ws://localhost:8765") await asyncio.Event().wait() if __name__ == "__main__": try: asyncio.run(main_async()) except KeyboardInterrupt: print("\n🛑 Gateway stopped by user.")
